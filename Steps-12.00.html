1. Create a new table and name it 'Measures Table'


2. Create new measures on the new table by copying and pasting .

.Total Participants = sum('InstLevel_Comprehensive'[EFTotalCount])
.Female Count % = DIVIDE(
SUM('InstLevel_Comprehensive'[EFFemaleCount]),
SUM('InstLevel_Comprehensive'[EFTotalCount])
)
.Male Exp to Female Exp = AVERAGE(InstLevel_Comprehensive[IL_SUM_OPEXPPERPART_MEN])/ AVERAGE(InstLevel_Comprehensive[IL_SUM_OPEXPPERPART_WOMEN])
Ratio Men STUaid to Women StuAid = DIVIDE(
(SUM('InstLevel_Comprehensive'[STUDENTAID_MEN])/SUM('InstLevel_Comprehensive'[EFMaleCount])),
(SUM('InstLevel_Comprehensive'[STUDENTAID_WOMEN])/SUM('InstLevel_Comprehensive'[EFFemaleCount])))

HEADCOACH MEN VS WOMEN PAY = DIVIDE(
SUM(InstLevel_Comprehensive[HDCOACH_SALARY_MEN]),
SUM(InstLevel_Comprehensive[HDCOACH_SALARY_WOMEN])
)
.Male Rev to Female Rev = DIVIDE (
SUM('InstLevel_Comprehensive'[IL_REVENUE_MENALL]),
SUM('InstLevel_Comprehensive'[IL_REVENUE_WOMENALL])
)
.Total Male Cnt = CALCULATE(SUMX('InstLevel_Comprehensive','InstLevel_Comprehensive'[EFMaleCount]))
.Total Female Cnt = CALCULATE(SUMX('InstLevel_Comprehensive','InstLevel_Comprehensive'[EFFemaleCount]))
Female Count % = DIVIDE([.Total Female Cnt], ([.Total Male Cnt]+ [.Total Female Cnt]))

.ArrowColorCount = IF([Female Count %]<50,"#A8337D","Dark Blue")
.ArrowColorExpense = IF([.Male Exp to Female Exp]>1,"Dark Blue","#A8337D")
.ArrowColorRevenue = IF([.Male Rev to Female Rev]>1,"Dark Blue","#A8337D")
.ArrowColorSalary = IF([HEADCOACH MEN VS WOMEN PAY]>1,"Dark Blue","Pink")
.ArrowColorStuAid = IF([Ratio Men STUaid to Women StuAid]>1,"Dark Blue","Pink")
.ArrowForCount = SWITCH(TRUE(),[Female Count %]<50,UNICHAR(9650),
[Female Count %]<50,UNICHAR(9660))
.ArrowForExpense = SWITCH(TRUE(),[.Male Exp to Female Exp]>1,UNICHAR(9660),
[.Male Exp to Female Exp]<1,UNICHAR(9650))
.ArrowForRevenue = SWITCH(TRUE(),[.Male Rev to Female Rev]>1,UNICHAR(9660),
[.Male Rev to Female Rev]<1,UNICHAR(9650))
.ArrowForSalary = SWITCH(TRUE(),[HEADCOACH MEN VS WOMEN PAY]<1,UNICHAR(9650),
[HEADCOACH MEN VS WOMEN PAY]>1,UNICHAR(9660))
.ArrowForStuAid = SWITCH(TRUE(),[Ratio Men STUaid to Women StuAid]<1,UNICHAR(9650),
[Ratio Men STUaid to Women StuAid]>1,UNICHAR(9660))
.Bubble Color = IF([HEADCOACH MEN VS WOMEN PAY]>1,"#0D6ABF","#A8337D")
.Bubble Revenue Color = IF([.Male Rev to Female Rev]>1,"#0D6ABF","#A8337D")
